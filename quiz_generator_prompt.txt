def process_question(q):
    if q["type"] == "single":
        opts = q["options"]
        correct_text = opts[0]
        rng.shuffle(opts)
        q["options"] = opts
        q["correctAnswer"] = opts.index(correct_text)

    elif q["type"] == "multiple":
        opts = q["options"]
        correct_indices = q["correctAnswer"] if isinstance(q["correctAnswer"], list) else [0]
        correct_items = [opts[i] for i in correct_indices]
        rng.shuffle(opts)
        q["options"] = opts
        q["correctAnswer"] = sorted([opts.index(c) for c in correct_items])

    elif q["type"] == "truefalse":
        # Do not shuffle true/false options to maintain standard ["True", "False"] formatting
        pass

            rng.shuffle(indices)
            q["items"] = [items[i] for i in indices]
            old_to_new = {old: new for new, old in enumerate(indices)}
            q["correctOrder"] = [old_to_new[i] for i in q["correctOrder"]]

        elif q["dragdropType"] == "match":

            items = q["items"]

            rng.shuffle(pairs)
            q["items"] = [p[0] for p in pairs]

            rng.shuffle(target_indices)
            old_to_new = {old: new for new, old in enumerate(target_indices)}
            q["targets"] = [targets[i] for i in target_indices]
            q["correctMapping"] = [old_to_new[m] for m in q["correctMapping"]]
